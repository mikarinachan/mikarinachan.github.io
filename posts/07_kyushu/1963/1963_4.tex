\begin{description}
\item[(a)]$P(x, \, y)$，$Q(u, \, v)$は$O$より出る同一半直線上にあって$OP \cdot OQ=1$
\item[(b)]$Q'$は$x$軸に関する$Q$の対称点
\item[(c)]$M$は線分$P'Q$の中点
\end{description}
このとき
\begin{description}
\item[(1)]$u$，$v$を$x$，$y$で表わせ。
\item[(2)]点$M$が$x$軸上を動くとき点$P$の軌跡を求めよ。
\end{description}
