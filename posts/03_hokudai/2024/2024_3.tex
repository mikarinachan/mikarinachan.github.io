\[ f_1(x)=3x \]
\[ f_{n+1}(x)=(n+2)x^{n+1}+\left( \int_0^1f_n(t)dt \right)x
\hspace{1zw} (n=1, \, 2, \, 3, \, \cdots\cdots) \]
