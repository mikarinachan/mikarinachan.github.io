\item[(2)]$f(x)$は実数全体を定義域とする微分できる関数で，$f(1)=0$である．このとき
\[ g(x)=\begin{cases} \displaystyle \frac{f(x)}{x-1} \hspace{1zw} (x\neq1\text{のとき}) \\
f'(1) \hspace{1zw} (x=1\text{のとき}) \end{cases} \]
とおけば，$g(x)$は連続関数である．
\end{description}
