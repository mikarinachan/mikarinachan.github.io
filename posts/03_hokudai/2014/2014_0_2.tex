　次の条件で定められる数列$\{a_n\}$を考える。
\[ a_1=1, \hspace{1zw} a_2=1, \hspace{1zw} a_{n+2}=a_{n+1}+3a_n \hspace{1zw}
(n=1, \, 2, \, 3, \, \cdots) \]
\begin{description}
\item[(1)]以下が成立するように，実数$s, \, t$ $(s>t)$を定めよ。
\[ \begin{cases} a_{n+2}-sa_{n+1}=t(a_{n+1}-sa_n) \\
a_{n+2}-ta_{n+1}=s(a_{n+1}-ta_n) \end{cases} \hspace{1zw}
(n=1, \, 2, \, 3, \, \cdots) \]
\item[(2)]一般項$a_n$を求めよ。
\end{description}
