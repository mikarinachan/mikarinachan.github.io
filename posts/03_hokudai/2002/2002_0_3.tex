\begin{description}
\item[(1)]次の連立不等式の表す領域を図示せよ．
\[ x+3y\leqq12, \hspace{1zw} 3x+y\leqq12, \hspace{1zw} a(x-3)+b(y-2)\leqq0 \hspace{1zw}
x\geqq0, \hspace{1zw} y\geqq0 \]
\item[(2)]実数$x$，$y$が(1)の連立不等式を満たすとき，$x+y$の最大値を$a$，$b$を用いて表せ．
\end{description}
