\begin{description}
\item[(1)]$a+b=c$であるとき，$a^3+b^3+3abc=c^3$が成り立つことを示せ．
\item[(2)]$a+b \geqq c$であるとき，$a^3+b^3+3abc \geqq c^3$が成り立つことを示せ．
\end{description}
