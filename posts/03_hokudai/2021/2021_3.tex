\[ \frac{9^{4x}+9^{y^2+1}}{6}=3^{4x+y^2} \hspace{1zw} \cdots\cdots\text{(★)} \]

\[ \frac{1}{\log_{1+\frac{x}{y}}4}+\frac{1}{\log_{1-\frac{x}{y}}4} \]

\end{description}
