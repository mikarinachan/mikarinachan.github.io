\begin{description}
\item[(1)]$f(x)$を$x$の整式とし，
$\{ a_k \}$は$a_k<a_{k+1}$ $(k=1, \, 2, \, \cdots\cdots)$
および$\displaystyle\lim_{k\to\infty}a_k=\infty$をみたす数列とする．
このとき$f(a_k)=0$，$k=1, \, 2, \, \cdots\cdots$ならば
$f(x)$は整式として0であることを示せ．
\item[(2)]$f_1(x)$，$f_2(x)$，$f_3(x)$を$x$の整式とし
$F(x)=f_1(x)+f_2(x)\sin x+f_3(x)\sin 2x$は
すべての実数$x$に対して0であるとする．
このとき$f_1(x)$，$f_2(x)$，$f_3(x)$はいずれも
整式として0であることを示せ．
\end{description}
