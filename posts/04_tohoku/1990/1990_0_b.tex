\[ C_1:y=-3x^2, \hspace{1zw} C_2:y=3x^2-18ax+12a^2 \]

\begin{description}
\item[(1)]$\displaystyle0<a\leqq\frac{1}{2}$のとき，$S$を$a$で表せ．
\item[(2)]$\displaystyle\frac{1}{2}<a<1$のとき，$S$を$a$で表せ．
\item[(3)]$0<a<1$における$S$の最大値と，そのときの$a$を求めよ．
\end{description}
