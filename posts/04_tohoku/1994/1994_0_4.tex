{\huge 4}
　2次関数$f_n(x)$ $(n=1, \, 2, \, \cdots\cdots)$を次のように定義する．
\[ f_1(x)=3x^2, \hspace{1zw} f_2(x)=3x^2+4x, \]
\[ f_n(x)=3x^2+4x\int_0^1f_{n-1}(t)dt-\int_0^1f_{n-2}(t)dt \hspace{1zw} (n\geqq3) \]
\begin{description}
\item[(1)]$a_1=0$，$\displaystyle a_n=\int_0^1f_{n-1}(t)dt$ $(n\geqq2)$とおくとき，
$a_n$を$a_{n-1}$と$a_{n-2}$で表す漸化式を求めよ．
\item[(2)]$f_n(x)$を求めよ．
\end{description}
\end{flushleft}
