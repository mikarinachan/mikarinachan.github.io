\[ a_1=3, \hspace{1zw} a_{k+1}=4a_k+3 \hspace{1zw} (k=1, \, 2, \, \cdots\cdots) \]

\item[(2)]次の和を求めよ．$\displaystyle\sum_{k=1}^n\frac{1}{\log_2(a_k+1)\log_2(a_{k+1}+1)}$
\end{description}
