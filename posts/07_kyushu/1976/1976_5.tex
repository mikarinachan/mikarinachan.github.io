\begin{description}

\item[(2)]$n$を偶数とし，$f(x)$の$n$乗を$a_0x^n+a_1x^{n-1}+\cdots\cdots+a_n$とおくとき
\[ {\left( \frac{3}{2} \right)}^n \geqq a_0+a_2+\cdots\cdots+a_{n-2}+a_n>\frac{1}{2}{\left( \frac{3}{2} \right)}^n \]

\end{description}
